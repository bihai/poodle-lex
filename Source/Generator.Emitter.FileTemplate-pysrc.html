23 -def FileTemplate(in_filename, out_filename): 

24      """ 
25      Helper class for filling in template files. 
26      Returns an iterator which copies a file's contents, pausing to yeild each time it hits a $VARIABLE token. 
27      Yields the following for each $VARIABLE token in the file: 
28          1. A Python file object representing the output file 
29          2. A string containing the name of the token 
30          3. If the token is the first non-whitespace of a line, an integer with the indentation, otherwise None. 
31      @param in_filename: string containing the name of the template file to copy 
32      @param out_filename: string containing the name of the filled-in template file to write. 
33      """ 
34      pattern_variablename = r"\$(?P<%s>[a-zA-Z][a-zA-Z0-9_]*)" 
35      pattern_inline = pattern_variablename % "inline" 
36      pattern_entireline = r"^(?P<whitespace>[ \t]*)%s" % (pattern_variablename % "entireline") 
37      pattern_variable = r"(%s)|(%s)" % (pattern_entireline, pattern_inline) 
38       
39      with open(in_filename, 'r') as in_file: 
40          with open(out_filename, 'w') as out_file: 
41              for line in in_file: 
42                  match = re.search(pattern_variable, line) 
43                  if match is not None: 
44                      if match.group('entireline') is not None: 
45                          yield out_file, match.group('entireline'), len(match.group('whitespace')) 
46                      else: 
47                          out_file.write(line[:match.start(0)]) 
48                          yield out_file, match.group('inline'), None 
49                          out_file.write(line[match.end(0):]) 
50                  else: 
51                      out_file.write(line)
